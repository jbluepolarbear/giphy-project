//create on click function for buttons created in html
$("button").on("click", function(){
  // The button will hold variables - the peron's name using attr("data person") and the API URL
  var person = $(this).attr("data-person");
      var queryURL = "https://api.giphy.com/v1/gifs/search?q=" +
        person + "&api_key=dc6zaTOxFJmzC&limit=10";
        // AJAX takes the URL and gets the data from the API
      $.ajax({
        url: queryURL,
        method: "GET"
      }) // upon loading the API info, create a response function that produces the data results
        .then(function(response) {
          var results = response.data;
          console.log(results);
        })
        // Loop through all of the gif results 
        for (var i = 0; i < results.length; i++) {
            var gifDiv = $("<div>");
            // Create a variable that is the rating of the gif
            var rating = results[i].rating;
            // Create a new paragraph that includes the rating
            var p = $("<p>").text("Rating: " + rating);
// Create a space in the HTML for the image
            var personImage = $("<img>");
            // attach the results from the API to the person image
            personImage.attr("src", results[i].images.fixed_height.url);

            gifDiv.prepend(p);
            gifDiv.prepend(personImage);

            $("#gif-view").prepend(gifDiv);
            }
})
