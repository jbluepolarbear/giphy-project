//function calls on giphy API and renders data from the array

function displayNameInfo() {
  var name = $(this).attr("data-name");
  console.log(name);
  var queryURL = "https://api.giphy.com/v1/gifs/search?q=" + name + "&api_key=EeLVOUPGYR0IFcYTnM4KxPnPy4CEKtkr&limit=10"; 
  //AJAX call for the specific name being clicked
  $.ajax({
    url: queryURL,
    method: "GET" 
    })
    .then(function(response) {
      console.log(response.data);
      //create a div to hold the gifs
      var gifDiv = $("<div class='gif'>");
      //store the rating data
      var rating = response.Rated;
      console.log(response.Rated);
      //create element to display rating
      var p = $("<p>").text("Rating: " + rating);
      //display the rating
      gifDiv.append(p)
      //retrieve the URL for the gif image
      var gifyURL = response.data.image_original_url;
    
      //create an element to hold theh image
      var image = $("<img>").attr("src", gifyURL);
      //append the image
      gifDiv.append(image);
      //put character gifs above the previous gifs
      $("gif-view").prepend(gifDiv);
    });

}
